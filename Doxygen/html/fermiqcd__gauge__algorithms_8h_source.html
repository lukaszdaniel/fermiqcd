218  unsigned short int mu;
219  site x(U.lattice());

234  for (int i = 0; i < U.nc; i++) {
235  for (int k = 0; k < i; k++) {
236  scalar = 0;
237  for (int j = 0; j < U.nc; j++) {
238  scalar += conj(e(k, j)) * e(i, j);
239  } //<ek|ei>
240  for (int j = 0; j < U.nc; j++) {
241  e(i, j) -= e(k, j) * scalar;
242  } //ei = ei - ek<ek|ei>
243  }
244  quadnorm = 0;
245  for (int j = 0; j < U.nc; j++)
246  quadnorm += abs2(e(i, j)); //quadnorm = ||ei||^2
247 
248  for (int j = 0; j < U.nc; j++)
249  e(i, j) /= sqrt(quadnorm); //ei = ei/||ei||
250  }
251 
252  //The last vector is calculated as the cross product: ek = ei x ej
253  i = U.nc - 1;
254  for (int j = 0; j < U.nc; j++) {
255  for (int s = 0; s < U.nc - 1; s++) {
256  for (int k = 0; k < U.nc - 1; k++) {
257  if (k < j)
258  m(s, k) = e(s, k);
259  else
260  m(s, k) = e(s, k + 1);
261  }
262  }

264  if ((j + i) % 2 == 1)
265  e(i, j) *= -1;
266  }
267  U(x, mu) = e;
268  }

405 mdp_matrix project_SU(mdp_matrix M, int nstep = 1) {
406  int i, j, k, l, step, nc = M.rowmax();
407  mdp_real e0, e1, e2, e3, dk, d;
408  mdp_complex dc, u0, u1, u2, u3;
409  mdp_matrix B(nc, nc);
410  mdp_matrix C(nc, nc);
411  mdp_matrix S(nc, nc);
412 
413  C = M;
414  // /////////////////
415  // preconditioning
416  // /////////////////
417  for (i = 0; i < nc; i++) {
418  for (j = 0; j < i; j++) {
419  dc = 0;
420  for (k = 0; k < nc; k++)
421  dc += conj(C(k, j)) * C(k, i);
422  for (k = 0; k < nc; k++)
423  C(k, i) -= dc * C(k, j);
424  }
425  d = 0;
426  for (k = 0; k < nc; k++)
427  d += pow((double) abs(C(k, i)), (double) 2.0);
428  d = sqrt(d);
429  for (k = 0; k < nc; k++)
430  C(k, i) /= d;
431  }
432  // ////////////////////////////
433  // Cabibbo Marinari Projection
434  // ////////////////////////////
435  for (i = 0; i < nc; i++)
436  for (j = 0; j < nc; j++)
437  for (k = 0; k < nc; k++)
438  B(i, j) += conj(M(k, i)) * C(k, j);
439  for (step = 0; step < nstep; step++) {
440  for (i = 0; i < nc - 1; i++)
441  for (j = i + 1; j < nc; j++) {
442  e0 = real(B(i, i)) + real(B(j, j));
443  e1 = imag(B(i, j)) + imag(B(j, i));
444  e2 = real(B(i, j)) - real(B(j, i));
445  e3 = imag(B(i, i)) - imag(B(j, j));
446  dk = sqrt(e0 * e0 + e1 * e1 + e2 * e2 + e3 * e3);
447  u0 = mdp_complex(e0, -e3) / dk;
448  u1 = mdp_complex(e2, -e1) / dk;
449  u2 = mdp_complex(-e2, -e1) / dk;
450  u3 = mdp_complex(e0, e3) / dk;
451  // S=C;
452  for (k = 0; k < nc; k++) {
453  S(k, i) = C(k, i) * u0 + C(k, j) * u1;
454  S(k, j) = C(k, i) * u2 + C(k, j) * u3;
455  }
456  if ((i == nc - 2) && (j == nc - 1))
457  for (k = 0; k < nc; k++)
458  for (l = 0; l < nc - 2; l++)
459  S(k, l) = C(k, l);
460  if ((i != nc - 2) || (j != nc - 1) || (step != nstep - 1))
461  for (k = 0; k < nc; k++) {
462  C(k, i) = B(k, i) * u0 + B(k, j) * u1;
463  C(k, j) = B(k, i) * u2 + B(k, j) * u3;
464  B(k, i) = C(k, i);
465  B(k, j) = C(k, j);
