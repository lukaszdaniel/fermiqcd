424  FermiClass &psol) {

427  int fnc = U.nc * (U.nc + 1) / 2;
428  mdp_matrix dum(U.nc, U.nc);
429  mdp_matrix tmp1(U.nc, U.nc);
430  mdp_matrix tmp2(U.nc, U.nc);
431  mdp_matrix stemp;
432  GaugeClass Utr(U);
433  GaugeClass Udag(U);
434  GaugeClass Udagtr(U);
435 
436  forallsites(x)
437  {
438  for (int mu = 0; mu < U.ndim; mu++) {
439  Udag(x, mu) = hermitian(U(x, mu));
440  Utr(x, mu) = transpose(U(x, mu));
441  Udagtr(x, mu) = transpose(Udag(x, mu));
442  }
443  }
444  Udag.update();
445  Utr.update();
446  Udagtr.update();
447 
448  if (coeff["representation"] == FUNDAMENTAL) {
449  forallsites(x)
450  {

452  dum = 0;
453  for (int a = 0; a < U.nc; a++) {
454  for (int b = 0; b < U.nc; b++) {
455  stemp =
456  0.5 * hermitian(spinor(psol, x, a))
457  * ((1 - Gamma[mu])
458  * spinor(sol, x + mu, b));
459  tmp1(b, a) = stemp(0, 0);
460  stemp = 0.5 * hermitian(spinor(psol, x + mu, a))
461  * ((1 + Gamma[mu]) * spinor(sol, x, b));
462  tmp2(b, a) = stemp(0, 0);
463  }
464  }
465  dum = tmp2 * Udag(x, mu) - U(x, mu) * tmp1;
466  dum -= hermitian(dum);
467  dum -= trace(dum) * mdp_identity(U.nc) / U.nc;
468  f_U(x, mu) = dum;
469  }
470  }
471  } else if (coeff["representation"] == SYMMETRIC) {
472  forallsites(x)
473  {
474  for (int mu = 0; mu < U.ndim; mu++) {
475  dum = 0;
476  for (int a = 0; a < fnc; a++) {
477  for (int b = 0; b < fnc; b++) {
478  stemp =
479  hermitian(spinor(psol, x, a))
480  * ((1 - Gamma[mu])
481  * spinor(sol, x + mu, b));
482  dum -= stemp(0, 0) * U(x, mu) * S[b] * Utr(x, mu)
483  * S[a];
484  stemp = hermitian(spinor(psol, x + mu, a))
485  * ((1 + Gamma[mu]) * spinor(sol, x, b));
486  dum += stemp(0, 0) * S[b] * Udagtr(x, mu) * S[a]
487  * Udag(x, mu);
488  }
489  }
490  dum -= hermitian(dum);
491  dum -= trace(dum) * mdp_identity(U.nc) / U.nc;
492  f_U(x, mu) = dum;
493  }
494  }
495  } else if (coeff["representation"] == SO4) {
496  mdp_real f;

498  {
499  for (int mu = 0; mu < U.ndim; mu++) {
500  dum = 0;

502  f = 2.0 * trace(lambda[g] * tmp2 * Udag(x, mu)).real()
503  - 2.0 * trace(lambda[g] * U(x, mu) * tmp1).real();
504  dum = dum - f * lambda[g];
505  }
506  f_U(x, mu) = dum;
507  }
508  }
