64  inline int ordered_index(int mu, int nu) const {
65  // ////////////////////////
66  // this map mu, nu -> k //
67  // 0 1 0 //
68  // 0 2 1 //
69  // 0 3 2 //
70  // 1 2 3 //
71  // 1 3 4 //
72  // 2 3 5 //
73  // It works for any ndim //
74  // ////////////////////////
75  if (mu < nu)
76  return nu + (mu * (2 * ndim - mu - 3)) / 2 - 1;
77  if (mu > nu)
78  return mu + (nu * (2 * ndim - nu - 3)) / 2 - 1 + nem;
79  // error("wrong call to ordered_index() with mu>=nu");

81  }
